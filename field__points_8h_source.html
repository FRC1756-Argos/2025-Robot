43 template <class T>
44 std::vector<T> ReflectFieldPoint(const std::vector<T> source) {
45 std::vector<T> retVal;
46 retVal.reserve(source.size());
47 std::transform(
48 source.begin(), source.end(), std::back_inserter(retVal), [](T val) { return ReflectFieldPoint(val); });
49 return retVal;
50 }
